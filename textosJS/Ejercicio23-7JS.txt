<h2>Mostrar cada palabra en una línea</h2>
<script>
let oracion = prompt("Ingresa una oración:");
let palabras = oracion.split(" ");

for (let i = 0; i < palabras.length; i++) {
  document.write(palabras[i] + "<br>");
}
</script>
